mac_xampp

###配置虚拟域名

subl /Applications/XAMPP/etc/extra/httpd-vhosts.conf

subl /etc/hosts

###xampp 启动apache

sudo /Applications/XAMPP/xamppfiles/bin/apachectl start

###xampp 启动mysql

sudo /Applications/XAMPP/xamppfiles/bin/mysql.server start
